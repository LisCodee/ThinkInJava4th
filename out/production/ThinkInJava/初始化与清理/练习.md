    }

        System.out.println("重载构造器被调用：" + s);

    }
}

### 5.
```
public class Dog {
    void bark(int x){
        System.out.println("barking");
    }
    void bark(double x){
        System.out.println("howling");
    }

    public static void main(String[] args) {
        Dog d = new Dog();
        d.bark(1);
        d.bark(0.1);
    }
}
```
